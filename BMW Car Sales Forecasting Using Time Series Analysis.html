Preprocessing the Data

In[6]:
import pandas as pd

# Convert the Date column to datetime format
df['Date'] = pd.to_datetime(df['Date'], format='%d-%m-%Y')

# Group by Date and sum Quantity Sold if multiple entries for the same date
df_daily_sales = df.groupby('Date')['Quantity Sold'].sum().reset_index()

Train-Test Split

In[9]:
train = df_daily_sales[:-30]  # Train on all data except the last 30 days
test = df_daily_sales[-30:]   # Use the last 30 days for testing

Using ARIMA Model for Forecasting

In[10]:
from statsmodels.tsa.arima.model import ARIMA

# Fit ARIMA model (example: p=5, d=1, q=0)
model = ARIMA(train['Quantity Sold'], order=(5, 1, 0))
model_fit = model.fit()

# Forecasting the next 30 days
forecast = model_fit.forecast(steps=30)

# Plot the forecast
plt.figure(figsize=(10, 6))
plt.plot(train.index, train['Quantity Sold'], label='Training Data')
plt.plot(test.index, test['Quantity Sold'], label='Test Data')
plt.plot(test.index, forecast, label='Forecast', color='red')
plt.title('Sales Forecast')
plt.xlabel('Date')
plt.ylabel('Quantity Sold')
plt.legend()
plt.show()
